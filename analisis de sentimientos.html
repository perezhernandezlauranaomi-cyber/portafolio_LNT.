"""
crear un frase larga con palabras repetidas donde exprese
un sentimiento positivo  pensamientos nagativos, analizar la frase completa eliminando
signos de puntuacion
,tokenizar la frase y rcorriendo la frase con for, con if contabilizar cuantas
palabras positivas y cuantas negativas se encontraron y de ahi determinar si la frase es positiva o negativa
"""
#funciones clave:replace, split,lower,for,if
#crear √±istas con las palabras para detectar el sentir de frase
palabraspositivas=["feliz","contento","alegre","maravilloso","bonito","bonita"]
palabrasnegativas=["triste","solo","aburrido","malo","deprimido","horrible","llorando"]

frase=("fue muy triste cuando fallecio mi conejito, me la pase llorando toda una semana.Lo que me puse feliz fue que quedo mi conejita y tenia que estar para ella")
tokens= frase.lower().split()
verificadas=[]
tokens_limpio =frase.replace(",","").replace(",","").replace(",","").lower().split()

positivas=0
negativas=0
for palabra in tokens:
    if palabra in palabraspositivas:
        positivas=positivas+1
    if palabra in palabrasnegativas:
        negativas=negativas+1

print("palabras negativas ",negativas)
print("palabras positivas ", positivas)

if positivas > negativas:
    print("la frase es positiva")
else:
    print("la frase es negativa")
